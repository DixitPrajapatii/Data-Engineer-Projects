-- Count the number of Atheletes from each Country
SELECT * FROM athletes;

SELECT NOC, COUNT(*) AS Total_Atheletes 
FROM athletes
GROUP BY NOC
ORDER BY Total_Atheletes DESC

-- Calculate the total medals won by each country
SELECT * FROM medals;

SELECT 
Team_NOC AS Country_Name,
SUM(Gold) AS Total_Gold,
SUM(Silver) AS Total_Silver,
SUM(Bronze) AS Total_Bronze 
FROM medals
GROUP BY Team_NOC
ORDER BY Total_Gold DESC;


-- Calculate the average number of entries by gender for each discipline;
SELECT * FROM entriesgender;

SELECT 
Discipline,
AVG(Female) AS Female_Total,
AVG(Male) AS Male_Total
FROM entriesgender
GROUP BY Discipline;
